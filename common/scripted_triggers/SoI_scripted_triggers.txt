has_great_power_DLC_trigger = {
	OR = {
		has_dlc = "Rights of Man"
		has_dlc = "Emperor"
	}
}

is_great_power_trigger = {
	if = {
		limit = {
			has_great_power_DLC_trigger = yes
		}
		is_great_power = yes
	}
	else = {
		is_subject = no
		NOT = {
			calc_true_if = {
				all_country = {
					is_subject = no
					total_development = PREV
				}
				amount = 8
			}
		}
	}
}

is_great_power_rank = { # parameters: int 'value', bool 'exact'
	if = {
		limit = {
			has_great_power_DLC_trigger = yes
		}
		if = {
			limit = {
				is_great_power = yes
			}
			variable_arithmetic_trigger = {
				custom_tooltip = is_[[exact]exact_]great_power_rank_tt
				
				export_to_variable = {
					which = desired_great_power_rank
					value = $value$
				}
				export_to_variable = {
					which = our_great_power_rank
					value = trigger_value:great_power_rank # <-- Requires DLC to work.
				}
				
				check_variable = {
					which = desired_great_power_rank
					which = our_great_power_rank
				}
				[[exact]
					is_variable_equal = {
						which = desired_great_power_rank
						which = our_great_power_rank
					}
				]
			}
		}
		else = {
			is_great_power = yes
		}
	}
	else = {
		is_great_power_trigger = yes
		NOT = {
			calc_true_if = {
				all_country = {
					is_subject = no
					total_development = PREV
				}
				amount = $value$
			}
		}
		[[exact]
			calc_true_if = {
				all_countries_including_self = {
					is_subject = no
					total_development = PREV
				}
				amount = $value$
			}
		]
	}
}

AI_wants_to_be_influenced_by_trigger = { # THIS == Target, $who$ == Great Power
	OR = {
		AND = {
			is_subject_of = $who$
			NOT = { liberty_desire = 50 }
		}
		AND = {
			OR = {
				ai_attitude = {
					who = $who$
					attitude = attitude_neutral
				}
				ai_attitude = {
					who = $who$
					attitude = attitude_friendly
				}
				ai_attitude = {
					who = $who$
					attitude = attitude_defensive
				}
				ai_attitude = {
					who = $who$
					attitude = attitude_allied
				}
			}
			NOT = {
				OR = {
					government = tribal
					government = native
					is_primitive = yes
				}
			}
			NOT = { isolationism = 3 }
			
			any_neighbor_country = { # Feeling threatened by any neighbor
				PREV = {
					ai_attitude = {
						who = PREV
						attitude = attitude_threatened
					}
				}
				NOT = { tag = $who$ }
			}
			
			if = { # Any province in home region is owned by '$who$'
				limit = {
					NOT = { is_year = 1700 }
					NOT = { has_global_flag = age_of_influence } # Set flag to enable early
				}
				capital_scope = {
					region_for_scope_province = {
						owned_by = $who$
					}
				}
			}
		}
	}
}

AI_wants_to_influence_trigger = { # THIS == Great Power, $who$ == Target
	OR = {
		overlord_of = $who$
		AND = {
			OR = {
				ai_attitude = {
					who = $who$
					attitude = attitude_friendly
				}
				ai_attitude = {
					who = $who$
					attitude = attitude_defensive
				}
				ai_attitude = {
					who = $who$
					attitude = attitude_protective
				}
				ai_attitude = {
					who = $who$
					attitude = attitude_domineering
				}
				ai_attitude = {
					who = $who$
					attitude = attitude_allied
				}
				AND = {
					has_diplomatic_influence = {
						who = $who$
						value = 20
					}
					NOT = {
						OR = {
							ai_attitude = {
								who = $who$
								attitude = attitude_rivalry
							}
							ai_attitude = {
								who = $who$
								attitude = attitude_hostile
							}
							ai_attitude = {
								who = $who$
								attitude = attitude_outraged
							}
						}
					}
				}
			}
			$who$ = {
				NOT = {
					OR = {
						government = tribal
						government = native
						is_primitive = yes
					}
				}
			}
			
			if = { # Any owned province is in home region of '$who$'
				limit = {
					NOT = { is_year = 1700 }
					NOT = { has_global_flag = age_of_influence } # Set flag to enable early
				}
				any_owned_province = {
					region_for_scope_province = {
						is_capital_of = $who$
					}
				}
			}
		}
	}
}
