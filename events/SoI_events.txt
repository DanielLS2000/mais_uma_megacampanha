# Embassy under Siege
country_event = {
	id = SoI.1
	title = SoI.1.t
	desc = SoI.1.d
	picture = COUNTRY_COLLAPSE_eventPicture
	
	is_triggered_only = yes
	
	#trigger = {
	#	any_subject_country = {
	#		is_subject_of_type = sphere_member
	#		has_country_modifier = lost_to_rebels
	#	}
	#}
	
	option = { # We must protect our interests, with force.
		name = SoI.1.ra
		if = {
			limit = {
				ai = yes
				is_great_power_trigger = yes
				is_at_war = no
			}
			declare_war_with_cb = {
				who = FROM
				casus_belli = cb_restore_sphere
			}
		}
		else = {
			add_casus_belli = {
				target = FROM
				type = cb_restore_sphere
				months = 12
			}
		}
	}
	
	option = { # We'll salvage what little influence we have left.
		name = SoI.1.rb
		add_prestige = -15
		add_diplomatic_influence = {
			who = FROM
			value = 10
		}
	}
}
