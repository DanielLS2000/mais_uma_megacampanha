# Prints " (In Sphere)" if scope is in the Sphere of given GP event target.
defined_text = {
	name = GetInSphere_GP_1
	random = no
	
	text = {
		localisation_key = string_in_sphere
		trigger = {
			has_saved_global_event_target = GP_1
			
			is_subject_of = event_target:GP_1
			is_subject_of_type = sphere_member
		}
	}
	text = {
		localisation_key = string_is_subject
		trigger = {
			has_saved_global_event_target = GP_1
			
			is_subject_of = event_target:GP_1
		}
	}
	text = {
		localisation_key = ""
	}
}
